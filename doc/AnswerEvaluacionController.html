def view
  if not @evaluacion.taken_by? current_user
    redirect_to action: :show, id: params[:id]
  else
    @user = current_user
    @date = @user.user_answers.where(question_id: @evaluacion.questions_ids).first.created_at
    render "respuestas/show"
  end
end
